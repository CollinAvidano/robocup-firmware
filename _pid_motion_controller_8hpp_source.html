101  std::array<int16_t, 4> run(const std::array<int16_t, 4>& encoderDeltas,
102  float dt, Eigen::Vector4d* errors = nullptr,
103  Eigen::Vector4d* wheelVelsOut = nullptr,
104  Eigen::Vector4d* targetWheelVelsOut = nullptr) {
105  // update control targets
106  // in the future, we can get the rotation angle soccer wants and
107  // directly command that
108  // as our target, or integrate the rotational velocities given to us to
109  // create the target.
110  // For now though, we only do an angle hold when soccer is not
111  // commanding rotational velocities
112  // (this should help with strafing quickly)
113  // target_rotation += _targetVel[2] * dt;
114 
115  // get sensor data
116  imu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);
117 
118  // convert sensor readings to mathematically valid values
119  Eigen::Vector4d wheelVels;
120  wheelVels << encoderDeltas[0], encoderDeltas[1], encoderDeltas[2],
121  encoderDeltas[3];
122  wheelVels *= 2.0 * M_PI / ENC_TICKS_PER_TURN / dt;
123 
124  auto bot_vel = RobotModel::get().WheelToBot * wheelVels;
125 
126  // we have two redundent sensor measurements for rotation
127  // 32.8 comes from data sheet, units are LSB / (deg/s)
128  float ang_vel_gyro = (gz / 32.8f) * M_PI / 180.0f;
129  float ang_vel_enc = bot_vel[2];
130 
131  // printf("%f\r\n", ang_vel_enc);
132  // std::printf("%f %f\r\n", ang_vel_gyro, ang_vel_enc);
133 
134  // perform sensor fusion
135  // the higher this is, the more gyro measurements are used instead of
136  // encoders
137  float sensor_fuse_ratio = 1;
138  float ang_vel_update = ang_vel_gyro * sensor_fuse_ratio +
139  ang_vel_enc * (1 - sensor_fuse_ratio);
140 
141  // perform state update based on fused value, passed through a low
142  // passed filter
143 
144  // so far noise on the gyro seems pretty minimal, that's why this filter
145  // is off
146  float alpha = 1.0; // 0->1 (higher means less filtering)
147  angular_vel = (alpha * ang_vel_update + (1 - alpha) * angular_vel);
148  // current rotation estimate
149  rotation += angular_vel * dt;
150 
151  // printf("%f\r\n", rotation * 180.0f / M_PI);
152 
153  // velocity we are actually basing control off of, not the latest
154  // command
155  auto target_vel_act = _targetVel;
156 
157  // std::printf("%f\r\n", rot_error);
158 
159  const auto epsilon = 0.0001f;
160  // soccer tells us to "halt" by sending 0 vel commands, we want to
161  // freeze the
162  // rotational controller too so bots dont' react when getting handled
163  bool soccer_stop = (std::abs(target_vel_act[0]) < epsilon) &&
164  (std::abs(target_vel_act[1]) < epsilon);
165  if (!soccer_stop && std::abs(target_vel_act[2]) < epsilon) {
166  if (!angle_hold) {
167  target_rotation = rotation;
168  angle_hold = true;
169  }
170 
171  // get the smallest difference between two angles
172  float rot_error = target_rotation - rotation;
173  while (rot_error < -M_PI) rot_error += 2 * M_PI;
174  while (rot_error > M_PI) rot_error -= 2 * M_PI;
175 
176  target_vel_act[2] = rotation_pid.run(rot_error);
177  } else {
178  // let target_vel_act be exactly what soccer commanded
179  angle_hold = false;
180  }
181 
182  // conversion to commanded wheel velocities
183  Eigen::Vector4d targetWheelVels =
184  RobotModel::get().BotToWheel * target_vel_act.cast<double>();
185 
186  if (targetWheelVelsOut) {
187  *targetWheelVelsOut = targetWheelVels;
188  }
189 
190  Eigen::Vector4d wheelVelErr = targetWheelVels - wheelVels;
191 
192  if (errors) {
193  *errors = wheelVelErr;
194  }
195 
196  if (wheelVelsOut) {
197  *wheelVelsOut = wheelVels;
198  }
199 
200  // Calculated by checking for slippage at max accel, and decreasing
201  // appropriately
202  // Binary search works really well in this case
203  // Caution: This is dependent on the PID values so increasing the
204  // agressiveness of that will change this
205  double max_error = 3.134765625;
206  double scale = 1;
207 
208  for (int i = 0; i < 4; i++) {
209  if (abs(wheelVelErr[i]) > max_error) {
210  scale = max(scale, abs(wheelVelErr[i]) / max_error);
211  }
212  }
213 
214  wheelVelErr /= scale;
215  targetWheelVels = wheelVels + wheelVelErr;
216 
217  std::array<int16_t, 4> dutyCycles;
218  for (int i = 0; i < 4; i++) {
219  float dc =
220  targetWheelVels[i] * RobotModel::get().DutyCycleMultiplier +
221  copysign(4, targetWheelVels[i]);
222 
223  dc += _controllers[i].run(wheelVelErr[i]);
224 
225  if (std::abs(dc) > FPGA::MAX_DUTY_CYCLE) {
226  // Limit to max duty cycle
227  dc = copysign(FPGA::MAX_DUTY_CYCLE, dc);
228  // Conditional integration indicating open loop control
229  _controllers[i].set_saturated(true);
230  } else {
231  _controllers[i].set_saturated(false);
232  }
233 
234  dutyCycles[i] = static_cast<int16_t>(dc);
235  }
236 
237  return dutyCycles;
238  }
